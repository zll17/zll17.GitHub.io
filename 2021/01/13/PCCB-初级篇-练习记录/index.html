    #include <cstdio>
using namespace std;

int dfs(char** mat,int H, int W, int posi, int posj){
    mat[posi][posj] = '@'; // label as used
    int di[4] = {-1,0,1,0};
    int dj[4] = {0,1,0,-1};
    int cnt = 1;
    for(int i=0;i<4;++i){
        int ci = posi + di[i];
        int cj = posj + dj[i];
        if(0<=ci && ci<H && 0<=cj && cj<W && mat[ci][cj]=='.'){
                cnt += dfs(mat,H,W,ci,cj);
        }
    } 
    return cnt;
}

int main() {
    int W,H;
    while(true){
        scanf("%d %d",&W,&H);
        if(W==0 && H==0){ break; }
        char **mat = new char*[H];
        int posi = 0, posj = 0;
        for(int i=0;i<H;++i){
            mat[i] = new char[W];
            for(int j=0;j<W;++j){ 
                scanf("%d ",&mat[i][j]);
                if(mat[i][j]=='@'){ posi=i; posj=j; }
            }
        }
        int res = dfs(mat,H,W,posi,posj);
        printf("%d\n",res);
    }
    return 0;
}

  • AOJ 0118: Property Distribution

    Code:

    #include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const int maxn = 100000 + 5;

void dfs(char **mat,int y,int x,int H,int W){
    int dy[4] = {-1,0,1,0};
    int dx[4] = {0,1,0,-1};
    char cf = mat[y][x];
    mat[y][x] = '.';
    for(int i=0;i<4;++i){
        int cy = y+dy[i], cx = x+dx[i];
        if(0<=cy && cy<H && 0<=cx && cx<W){
            if(mat[cy][cx]==cf){
                dfs(mat,cy,cx,H,W);
            }
        }
    }
}

int main() {
    int H,W;
    while(cin >> H >> W){
        if(H==0 && W==0){ break; }
        char **mat = new char*[H];
        for(int i=0;i<H;++i){
            mat[i] = new char[W];
            for(int j=0;j<W;++j){
                cin >> mat[i][j];
            }
        }
        int cnt=0;
        for(int i=0;i<H;++i){
            for(int j=0;j<W;++j){
                if(mat[i][j]!='.'){
                    ++cnt;
                    dfs(mat,i,j,H,W);
                }
            }
        }
        cout << cnt << endl;
    }
    return 0;
}

    #include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const int maxn = 100000 + 5;

int l = 0, r=0, flag = 0;
int arr[11];

void dfs(int t){
    if(t==10){
        flag = 1;
        return;
    }
    if(arr[t]>l){
        int black = l;
        l = arr[t];
        dfs(t+1);
        l = black;
    }
    if(arr[t]>r){
        int black = r;
        r = arr[t];
        dfs(t+1);
        r = black;
    }
}

int main() {
    int N;
    cin >> N;
    while(N--){
        for(int i=0;i<10;++i){
            cin >> arr[i];
        }
        flag = 0;
        dfs(0);
        if(flag == 1){ cout << "YES" << endl;}
        else{ cout << "NO" << endl; }
    }
    
    return 0;
}
