  • 排列枚举

    #include <algorithm>
#include <iostream>
using namespace std;

const int MAX_N = 10;
bool visited[MAX_N];
int perm1[MAX_N];

void print(int *ipt,int n){
    for(int i=0;i<n;++i){
        cout << ipt[i] << " ";
    }
    cout << endl;
}


void permutation1(int pos, int n){
    if(pos==n){
        print(perm1,n);
        return;
    }
    for(int i=0;i<n;++i){
        if(!visited[i]){
            visited[i] = true;
            perm1[pos] = i;
            permutation1(pos+1,n);
            visited[i] = false;
        }
    }
}

int perm2[MAX_N];

void permutation2(int n){
    for(int i=0;i<n;++i){
        perm2[i]=i;
    }
    do{
        print(perm2,n);
    }while(next_permutation(perm2,perm2+n));
    return;
}

int main(){
    int n;
    cout << "Input a number:" << endl;
    cin >> n;
    permutation1(0,n);
    return 0;
}

  • Saruman’s Army

    #include <iostream>
#include <algorithm>
using namespace std;

int solve(int R, int N, int *Xpt){
   sort(Xpt,Xpt+N);
   int cnt=0;
   int *pos=Xpt;
   while((pos-Xpt)<=N-1){
       pos = upper_bound(pos,Xpt+N,*pos+R);
       --pos;
       pos = upper_bound(pos,Xpt+N,*pos+R);
       ++cnt;
   }
   return cnt;
}

int main(){
    int R,N;
    int *Xpt;
    while(true){
        cin >> R >> N;
        if(R<0 || N<0){ break; }
        Xpt = new int[N]();
        for(int i=0;i<N;++i){ cin >> Xpt[i]; }
        int res = solve(R,N,Xpt);
        cout << res << endl;
    }
    return 0;
}

  • Fence Repair (POJ3253)

    #include <iostream>
#include <algorithm>
using namespace std;

int main(){
    int N;
    int *ptrL;
    while(cin>>N){
        if(N<=0){ cout << 0 << endl; continue;}
        ptrL = new int[N]();
        for(int i=0;i<N;++i){ cin >> ptrL[i]; }
        sort(ptrL,ptrL+N);
        int res = N>1 ? ptrL[0]+ptrL[1] : ptrL[0];
        int cur_cost = res;
        for(int i=2;i<N;++i){
            cur_cost += ptrL[i];
            res += cur_cost;
        }
        cout << res << endl;
    }
}

  • 0-1背包问题

    #include <iostream>
#include <cstring>
using namespace std;

int main(){
    int N, W;
    while(cin>> N >> W){
        int *v = new int[N]();
        int *w = new int[N](); 
        for(int i=0;i<N;++i){ cin >> w[i]; }
        for(int i=0;i<N;++i){ cin >> v[i]; }
        int **dp = new int*[N+1]();
        for(int i=0;i<N+1;++i){
            dp[i] = new int[W+1](); 
            for(int j=0;j<W+1;++j){
                dp[i][j] = 0;
            }
        }
        // dp[0][0...W]=0, dp[1..N][0]=0
        for(int i=1;i<N+1;++i){
            for(int j=1;j<W+1;++j){
                if(w[i-1]>j){ dp[i][j] = dp[i-1][j]; }
                else{
                    dp[i][j] = max(dp[i-1][j],dp[i-1][j-w[i-1]]+v[i-1]);
                }
            }
        }
        cout << dp[N][W] << endl;
    }
    return 0;
}
